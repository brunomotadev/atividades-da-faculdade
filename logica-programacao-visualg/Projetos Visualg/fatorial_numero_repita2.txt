var
    contador, numero, fatorial: Inteiro
    resposta: Caractere
inicio
Repita
    Escreva("Digite um numero: ")
    Leia (numero)
    contador <- numero
    fatorial <- 1
    Repita
          fatorial <- fatorial * contador
          contador <- contador - 1
    Ate ( contador < 1)
    EscrevaL ("O valor do fatorial de ", numero, " é igual a ", fatorial)
    Escreva("Quer continuar? [s/n]")
    Leia(resposta)
    LimpaTela
Ate (resposta = "n")
