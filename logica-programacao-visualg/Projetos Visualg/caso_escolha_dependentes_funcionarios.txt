   nome: Caractere

   Dep: Inteiro

      Leia (sal)
      Escreva ("Qual é a quantidade de dependentes? ")
      Leia (dep)

              Caso 0
                   NSal <- Sal + (Sal*5/100)
              Caso 1, 2, 3

              Caso 4, 5, 6
                   NSal <- Sal + (Sal*15/100)
              OutroCaso
                   NSal <- Sal + (Sal*18/100)
      FimEscolha
      Escreval ("O novo salario de ", nome," sera de R$", NSal:5:2)
