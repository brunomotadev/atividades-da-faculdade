var
   c, n, contdiv: Inteiro
inicio
// Seção de Comandos
   c <- 1
   contdiv <- 0
   Escreva("Digite um número: ")
   Leia(n)
   Repita
   

         Se (n % c = 0) entao
             contdiv <- contdiv + 1
         FimSe
         c <- c+1
   Ate (c > n)
   Se (contdiv > 2) entao
      EscrevaL("O valor ", n, " não é primo!")
   senao
        EscrevaL("O valor ", n, " é primo!")
   FimSe
fimalgoritmo
