var
       n, s: Inteiro
       resp: Caractere
inicio
// Seção de Comandos 
   s <- 0


   Repita
            Escreva ("Digite o valor ==> ")
            Leia (n)
            s <- s+n
            Escreva("Você quer continuar? [s/n]")
            Leia (resp)
   Ate (resp = "n")
   EscrevaL("A soma de todos os valores digitados é ", s)
